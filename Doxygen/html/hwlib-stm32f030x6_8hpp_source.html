150 public:

152  volatile uint32_t & config_word;
153  uint32_t pin;

157  void config( uint32_t conf ){
158  config_word &= ~( 0xF << config_offset );
159  config_word |= conf << config_offset;
160  }
161 
162  pin_base( uint32_t port_number, uint32_t pin_number, uint32_t conf ):
163  port{ port_registers( port_number ) },
164  config_word{ ( pin_number < 8 ) ? port.CRL : port.CRH },
165  pin{ pin_number },
166  config_offset{ 4 * ( pin_number % 8 ) },
167  mask{ 0x1U << pin_number }
168  {
169  config( conf );
170  }
171 
172  bool base_read(){
173  return (( port.IDR & mask ) != 0 );
174  }
175 
176  void base_write( bool v ){
177  port.BSRR |= ( v ? mask : ( mask << 16 ));
178  }

341 uint_fast64_t HWLIB_WEAK now_ticks(){
342 
343  static bool init_done = false;
344  if( ! init_done ){
345 /*
346  // switch to the 72 MHz crystal/PLL clock, from stm32x.cpp,
347  // some values taken from
348  // https://github.com/rogerclarkmelbourne/STM32duino-bootloader
349 
350  // Flash 2 wait state
351  FLASH->ACR &= (uint32_t)((uint32_t)~FLASH_ACR_LATENCY);
352  FLASH->ACR |= (uint32_t)FLASH_ACR_LATENCY_2;
353 
354  // Enable Prefetch Buffer
355  FLASH->ACR |= FLASH_ACR_PRFTBE;
356 
357  // enable HSE and wait for it
358  RCC->CR |= RCC_CR_HSEON;
359  while((RCC->CR & RCC_CR_HSERDY ) == 0 ){}
360 
361 
362  // PLL configuration: PLLCLK = HSE * 9 = 72 MHz
363  RCC->CFGR &= (uint32_t)((uint32_t)~(
364  RCC_CFGR_PLLSRC | RCC_CFGR_PLLXTPRE | RCC_CFGR_PLLMULL ));
365  RCC->CFGR |= (uint32_t)( RCC_CFGR_PLLSRC | RCC_CFGR_PLLMULL9 );
366 
367  // Enable PLL and wait for it
368  RCC->CR |= RCC_CR_PLLON;
369  while((RCC->CR & RCC_CR_PLLRDY) == 0){}
370 
371  // HCLK = SYSCLK
372  RCC->CFGR |= (uint32_t)RCC_CFGR_HPRE_DIV1;
373 
374  // PCLK2 = HCLK
375  RCC->CFGR |= (uint32_t)RCC_CFGR_PPRE2_DIV1;
376 
377  // PCLK1 = HCLK / 2
378  RCC->CFGR |= (uint32_t)RCC_CFGR_PPRE1_DIV2;
379 
380  // Select PLL as system clock source
381  RCC->CFGR &= (uint32_t)((uint32_t)~(RCC_CFGR_SW));
382  RCC->CFGR |= (uint32_t)RCC_CFGR_SW_PLL;
383 
384  // Wait till PLL is used as system clock source
385  while ((RCC->CFGR & (uint32_t)RCC_CFGR_SWS) != (uint32_t)0x08){}
386 */
387  // start the systick timer
388  SysTick->CTRL = 0; // stop the timer
389  SysTick->LOAD = 0xFFFFFF; // use its as a 24-bit timer
390  SysTick->VAL = 0; // clear the timer
391  SysTick->CTRL = 5; // start the timer, 1:1
392 
393  init_done = true;
394  }
395 
396  static unsigned int last_low = 0;
397  static unsigned long long int high = 0;
398 
399  // the timer ticks down, but we want an up counter
400  unsigned int low = 0xFFFFFF - ( SysTick->VAL & 0xFFFFFF );
401  if( low < last_low ){
402 
403  // the timer rolled over, so increment the high part
404  high += 0x1ULL << 24;
405  }
406  last_low = low;
407 
408  // return the aggregated ticks value
409  // the counter runs at 84 MHz
410  return ( low | high );
411 
412 }
